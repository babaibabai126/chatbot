   118→// ─── Gemini with Function Calling ───────────────────────────────────────
   119→async function callGemini(
   120→  messages: Array<{ role: string; content: string }>,
   121→  useTools: boolean = true
   122→): Promise<{ text: string | null; functionCall: { name: string; args: Record<string, unknown> } | null }> {
   123→  const apiKey = process.env.GEMINI_API_KEY;
   124→  if (!apiKey) return { text: null, functionCall: null };
   125→
   126→  try {
   127→    const contents = [];
   128→    for (let i = 1; i < messages.length; i++) {
   129→      const m = messages[i];
   130→      if (m.role === 'user') {
   131→        contents.push({ role: 'user', parts: [{ text: m.content }] });
   132→      } else if (m.role === 'assistant') {
   133→        contents.push({ role: 'model', parts: [{ text: m.content }] });
   134→      } else if (m.role === 'function_response') {
   135→        const parts = m.content.split('|||');
   136→        contents.push({
   137→          role: 'function',
   138→          parts: [{ functionResponse: { name: parts[0] || 'unknown', response: { result: parts[1] || '' } } }],
   139→        });
   140→      } else if (m.role === 'function_call') {
   141→        contents.push({ role: 'model', parts: [{ functionCall: JSON.parse(m.content) }] });
   142→      }
   143→    }
   144→
   145→    if (contents.length === 0) return { text: null, functionCall: null };
   146→
   147→    const body: Record<string, unknown> = {
   148→      contents,
   149→      systemInstruction: { parts: [{ text: SYSTEM_PROMPT }] },
   150→      generationConfig: { temperature: 0.3, maxOutputTokens: 2048 },
   151→    };
   152→
   153→    if (useTools) {
   154→      body.tools = [{ functionDeclarations: GEMINI_FUNCTIONS }];
   155→    }
   156→
   157→    const controller = new AbortController();
   158→    const timeout = setTimeout(() => controller.abort(), 25000);
   159→
   160→    const response = await fetch(
   161→      `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.0-flash:generateContent?key=${apiKey}`,
   162→      { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify(body), signal: controller.signal }
   163→    );
   164→
   165→    clearTimeout(timeout);
   166→    if (!response.ok) {
   167→      const errText = await response.text().catch(() => '');
   168→      console.error('[Chat] Gemini HTTP error:', response.status, errText.substring(0, 200));
   169→      return { text: null, functionCall: null };
   170→    }
   171→
   172→    const data = await response.json();
   173→    const candidate = data?.candidates?.[0];
   174→    if (!candidate?.content?.parts) return { text: null, functionCall: null };
   175→
   176→    let textResult: string | null = null;
   177→    let fcResult: { name: string; args: Record<string, unknown> } | null = null;
   178→
   179→    for (const part of candidate.content.parts) {
   180→      if (part.text) textResult = (textResult || '') + part.text;
   181→      if (part.functionCall) fcResult = { name: part.functionCall.name, args: part.functionCall.args || {} };
   182→    }
   183→
   184→    return { text: textResult, functionCall: fcResult };
   185→  } catch (err) {
   186→    console.error('[Chat] Gemini exception:', err);
   187→    return { text: null, functionCall: null };
   188→  }
   189→}

   235→function detectIntent(message: string): DetectedIntent | null {
   236→  const lower = message.toLowerCase().trim();
   237→
   238→  // ── Create Bill / Invoice ──
   239→  if (lower.match(/(create|make|generate|নতুন|তৈরি)\s*(an?\s*)?(gst\s*)?(bill|invoice|বিল|চালান)/) ||
   240→      lower.match(/(bill|invoice|বিল|চালান)\s*(create|make|generate|নতুন|তৈরি)/) ||
   241→      lower.match(/(create|make|generate)\s*(an?\s*)?(gst|non.?gst)\s*(bill|invoice)/)) {
   242→    const args: Record<string, unknown> = {};
   243→
   244→    // Detect bill type
   245→    if (lower.includes('gst') && !lower.includes('non-gst') && !lower.includes('non gst')) {
   246→      args.billType = 'gst';
   247→    } else if (lower.includes('non-gst') || lower.includes('non gst')) {
   248→      args.billType = 'non_gst';
   249→    } else if (lower.includes('gst')) {
   250→      args.billType = 'gst';
   251→    } else {
   252→      args.billType = 'non_gst'; // default
   253→    }
   254→
   255→    // Extract amount - look for ₹ or rs or inr patterns
   256→    const amountMatch = lower.match(/[₹rs\s]*([\d,]+(?:\.\d+)?)/);

   258→      const amount = parseFloat(amountMatch[1].replace(/,/g, ''));

   384→function extractClientName(message: string): string {
   385→  // Try patterns: "for X", "client X", "named X", "নাম X"
   386→  const patterns = [
   387→    /(?:for|client|customer|নাম)\s*:?\s*([A-Z][A-Za-z\s]{2,30}?)(?:\s*[,;.]|\s*(?:item|service|phone|email|with|amount|bill|create|add|₹)|$)/i,
   388→    /(?:named?|name)\s*:?\s*([A-Z][A-Za-z\s]{2,30}?)(?:\s*[,;.]|\s*(?:phone|email|with|item|₹)|$)/i,
   389→  ];
   390→
   391→  for (const p of patterns) {
   392→    const match = message.match(p);
   393→    if (match) return match[1].trim();
   394→  }
   395→  return '';
   396→}

   398→function extractItemDescription(message: string): string {
   399→  // Try to find item/service description
   400→  const patterns = [
   401→    /(?:item|service|product|আইটেম|সেবা)\s*(?:is|:)?\s*([A-Za-z\s&]{3,50}?)(?:\s*[,;. ]|$)/i,
   402→    /(?:for)\s+(?:[\w\s]+\s+)?(?:₹|rs\.?)/i, // "for Digital Marketing ₹10,000"
   403→  ];
   404→
   405→  for (const p of patterns) {
   406→    const match = message.match(p);
   407→    if (match && match[1]) return match[1].trim();
   408→  }
   409→
   410→  // Try to extract description between "for" and the amount
   411→  const forMatch = message.match(/for\s+(.+?)\s+(?:₹|rs\.?|of)/i);
   412→  if (forMatch) return forMatch[1].trim();
   413→
   414→  return '';
   415→}

   421→async function executeCreateBill(args: Record<string, unknown>, businessId: string): Promise<{ success: boolean; result: unknown; message: string }> {

   427→
   428→    // Find or create client
   429→    let clientId = '';
   430→    let clientObj = null;
   431→    if (clientName) {
   432→      clientObj = await db.client.findFirst({
   433→        where: { businessId, name: { contains: clientName, mode: 'insensitive' } },

   437→          data: { businessId, name: clientName, phone: String(args.clientPhone || '0000000000'), address: args.clientAddress ? String(args.clientAddress) : null },

   439→      }

   443→      if (firstClient) { clientId = firstClient.id; clientObj = firstClient; }
   444→    }
   445→
   446→    if (!clientId) {
   447→      return { success: false, result: null, message: '❌ কোনো ক্লায়েন্ট পাওয়া যায়নি। অনুগ্রহ করে ক্লায়েন্টের নাম দিন। / No client found. Please provide a client name.' };
   448→    }
   449→
   450→    if (items.length === 0) {
   451→      return { success: false, result: null, message: '❌ কোনো আইটেম পাওয়া যায়নি। অনুগ্রহ করে আইটেমের নাম ও মূল্য দিন। / No items found. Please provide item details.' };
   452→    }
   453→
   454→    let subtotal = 0;
   455→    let totalGst = 0;
   456→    const processedItems = items.map((item) => {
   457→      const rate = Number(item.rate) || 0;
   458→      const qty = Number(item.quantity) || 1;
   459→      const taxMode = String(item.taxMode || 'excl');
   460→      const baseRate = taxMode === 'incl' ? rate / (1 + GST_RATE) : rate;
   461→      const itemGst = billType === 'gst' ? baseRate * GST_RATE * qty : 0;
   462→      const amount = billType === 'gst'
   463→        ? (taxMode === 'incl' ? rate * qty : baseRate * (1 + GST_RATE) * qty)
   464→        : rate * qty;
   465→      subtotal += baseRate * qty;
   466→      totalGst += itemGst;
   467→      return {
   468→        description: String(item.itemName || ''),
   469→        quantity: qty, rate, amount,
   470→        itemName: String(item.itemName || ''),
   471→        taxMode,
   472→        cgst: billType === 'gst' ? baseRate * 0.09 * qty : 0,
   473→        sgst: billType === 'gst' ? baseRate * 0.09 * qty : 0,
   474→        baseRate,
   475→      };
   476→    });
   477→
   478→    const total = billType === 'gst' ? subtotal + totalGst : subtotal;
   479→
   480→    const bill = await db.bill.create({
   481→      data: {
   482→        businessId, clientId,
   483→        billNumber: `ATS/${new Date().getFullYear().toString().slice(2)}-${String(new Date().getMonth() + 1).padStart(2, '0')}/${String(Math.floor(Math.random() * 900) + 100)}`,
   484→        date: new Date(),
   485→        dueDate: new Date(Date.now() + 30 * 24 * 60 * 60 * 1000),
   486→        subtotal: billType === 'gst' ? subtotal : total,
   487→        tax: totalGst, discount: 0, total, billType,
   488→        clientGst: args.clientGst ? String(args.clientGst) : null,
   489→        clientAddress: args.clientAddress ? String(args.clientAddress) : null,
   490→        notes: args.notes ? String(args.notes) : null,
   491→        items: { create: processedItems },
   492→      },
   493→      include: { items: true, client: true },
   494→    });
   495→
   496→    const itemSummary = processedItems.map(i => `  • ${i.itemName}: ₹${i.rate} × ${i.quantity} = ₹${i.amount.toFixed(2)}`).join('\n');
   497→
   498→    return {
   499→      success: true, result: bill,
   500→      message: `✅ বিল তৈরি হয়েছে! / Bill Created!\n\n📄 Bill #${bill.billNumber}\n👤 Client: ${bill.client.name}\n📋 Items:\n${itemSummary}\n${billType === 'gst' ? `💰 Subtotal: ₹${subtotal.toFixed(2)}\n📊 GST (CGST 9% + SGST 9%): ₹${totalGst.toFixed(2)}\n` : ''}💵 Total: ₹${total.toFixed(2)}\n🏷️ Type: ${billType === 'gst' ?    617→async function executeGetDashboard(args: Record<string, unknown>, businessId: string): Promise<{ success: boolean; result: unknown; message: string }> {
   618→  try {
   619→    const period = String(args.period || 'month');
   620→    const now = new Date();
   621→    let startDate = new Date(now);
   622→    switch (period) {
   623→      case 'today': startDate = new Date(now.getFullYear(), now.getMonth(), now.getDate()); break;
   624→      case 'week': startDate = new Date(now.getTime() - 7 * 24 * 60 * 60 * 1000); break;
   625→      case 'month': startDate = new Date(now.getFullYear(), now.getMonth(), 1); break;
   626→      case 'year': startDate = new Date(now.getFullYear(), 0, 1); break;
   627→      default: startDate = new Date(2020, 0, 1);
   628→    }
   629→
   630→    const [bills, expenses, payments, clients] = await Promise.all([
   631→      db.bill.findMany({ where: { businessId, date: { gte: startDate } } }),
   632→      db.expense.findMany({ where: { businessId, date: { gte: startDate } } }),
   633→      db.payment.findMany({ where: { businessId, date: { gte: startDate } } }),
   634→      db.client.count({ where: { businessId } }),
   635→    ]);
   636→
   637→    const totalRevenue = bills.reduce((s, b) => s + b.total, 0);
   638→    const totalPaid = bills.reduce((s, b) => s + b.paidAmount, 0);
   639→    const totalExpenses = expenses.reduce((s, e) => s + e.amount, 0);
   640→    const totalPaymentsReceived = payments.reduce((s, p) => s + p.amount, 0);
   641→    const totalDue = totalRevenue - totalPaid;
   642→    const profit = totalPaid - totalExpenses;
   643→
   644→    const periodLabel: Record<string, string> = { today: 'আজ / Today', week: 'এই সপ্তাহ / This Week', month: 'এই মাস / This Month', year: 'এই বছর / This Year', all: 'সব / All Time' };
   645→
   646→    return {
   647→      success: true, result: { totalRevenue, totalPaid, totalExpenses, totalDue, profit, clients },
   648→      message: `📊 ${periodLabel[period] || periodLabel.month} সারসংক্ষেপ / Summary:\n\n💰 Revenue: ₹${totalRevenue.toFixed(2)}\n💵 Received: ₹${totalPaymentsReceived.toFixed(2)}\n📂 Expenses: ₹${totalExpenses.toFixed(2)}\n⚠️ Outstanding: ₹${totalDue.toFixed(2)}\n📈 Profit: ₹${profit.toFixed(2)}\n👥 Clients: ${clients}`,
   649→    };
   650→  } catch (error) {
   651→    return { success: false, result: null, message: `❌ Error: ${error instanceof Error ? error.message : 'Unknown'}` };
   652→  }

   709→export async function POST(request: NextRequest) {
   710→  try {
   711→    const { sessionId, message, context } = await request.json();
   712→    if (!message || typeof message !== 'string') {
   713→      return NextResponse.json({ error: 'Message is required' }, { status: 400 });
   714→    }
   715→
   716→    const businessId = context?.businessId || '';
   717→
   718→    // Build DB context
   719→    let dbContext = '';
   720→    if (businessId) {
   721→      try {
   722→        const [clients, recentBills] = await Promise.all([
   723→          db.client.findMany({ where: { businessId }, take: 20, select: { id: true, name: true, phone: true } }),
   724→          db.bill.findMany({ where: { businessId }, take: 5, include: { client: { select: { name: true } } }, orderBy: { createdAt: 'desc' } }),
   725→        ]);
   726→        dbContext = `\n\nCurrent Data:\nClients: ${JSON.stringify(clients.map(c => ({ id: c.id, name: c.name, phone: c.phone })))}\nRecent Bills: ${JSON.stringify(recentBills.map(b => ({ id: b.id, billNumber: b.billNumber, client: b.client.name, total: b.total, status: b.status, billType: b.billType })))}`;
   727→      } catch { /* ignore */ }
   728→    }
   729→
   730→    // Get or create conversation
   731→    let history = conversations.get(sessionId);
   732→    if (!history) {
   733→      history = [{ role: 'assistant', content: SYSTEM_PROMPT + dbContext }];
   734→    } else {
   735→      history[0] = { role: 'assistant', content: SYSTEM_PROMPT + dbContext };
   736→    }
   737→    history.push({ role: 'user', content: message });
   738→    if (history.length > 22) history = [history[0], ...history.slice(-20)];
   739→
   740→    let aiResponse: string | null = null;
   741→    let actionExecuted = false;
   742→    let actionData = null;
   743→
   744→    // ── STEP 1: Try intent detection (works without LLM!) ──
   745→    const intent = detectIntent(message);
   746→
   747→    if (intent && businessId) {
   748→      console.log(`[Chat] Detected intent: ${intent.action}`, JSON.stringify(intent.args));
   749→
   750→      // Try Gemini function calling first (for more accurate parameter extraction)
   751→      const geminiResult = await callGemini(history, true);
   752→
   753→      if (geminiResult.functionCall) {
   754→        // Gemini returned a function call - use it (more accurate params)
   755→        const result = await executeFunction(geminiResult.functionCall.name, geminiResult.functionCall.args, businessId);
   756→        actionExecuted = result.success;
   757→        actionData = result.success ? result.result : null;
   758→
   759→        if (result.success) {
   760→          // Send result back to Gemini for natural language formatting
   761→          history.push({ role: 'function_call', content: JSON.stringify({ name: geminiResult.functionCall.name, args: geminiResult.functionCall.args }) });
   762→          history.push({ role: 'function_response', content: `${geminiResult.functionCall.name}|||${result.message}` });
   763→
   764→          const followUp = await callGemini(history, false);
   765→          aiResponse = followUp.text || result.message;
   766→        } else {
   767→          aiResponse = result.message;
   768→        }
   769→      } else if (geminiResult.text) {
   770→        // Gemini responded with text but we have a detected intent
   771→        // Execute the action anyway using our detected params
   772→        const result = await executeFunction(intent.action, intent.args, businessId);
   773→        actionExecuted = result.success;
   774→        actionData = result.success ? result.result : null;
   775→
   776→        if (result.success) {
   777→          aiResponse = result.message;
   778→        } else {
   779→          aiResponse = geminiResult.text;
   780→        }
   781→      } else {
   782→        // Gemini failed - use our detected intent
   783→        const result = await executeFunction(intent.action, intent.args, businessId);
   784→        actionExecuted = result.success;
   785→        actionData = result.success ? result.result : null;
   786→        aiResponse = result.message;
   787→      }
   788→    } else if (!intent) {
   789→      // ── STEP 2: No intent detected - use LLM for conversation ──
   790→      const geminiResult = await callGemini(history, false);
   791→
   792→      if (geminiResult.text) {
   793→        aiResponse = geminiResult.text;
   794→      }
   795→
   796→      // Fallback to ZAI SDK
   797→      if (!aiResponse) {
   798→        try {
   799→          const zai = await getZAI();
   800→          if (zai) {
   801→            const zaiMessages = history
   802→              .filter(m => m.role === 'user' || (m.role === 'assistant' && m !== history![0]))
   803→              .slice(-10)
   804→              .map(m => ({ role: m.role === 'user' ? 'user' as const : 'assistant' as const, content: m.content }));
   805→            zaiMessages.unshift({ role: 'user', content: SYSTEM_PROMPT });
   806→
   807→            const zaiResponse = await zai.chat.completions.create({
   808→              messages: zaiMessages,
   809→              thinking: { type: 'disabled' },
   810→            });
   811→            aiResponse = zaiResponse?.choices?.[0]?.message?.content || null;
   812→          }
   813→        } catch { /* ignore */ }
   814→      }
   815→
   816→      // Smart fallback
   817→      if (!aiResponse) {
   818→        const lower = message.toLowerCase();
   819→        if (lower.match(/হ্যালো|হ্যাই|hello|hi|hey|নমস্কার|কেমন আছ/)) {

   823→        }
   824→      }
   825→    } else {
   826→      // Intent detected but no businessId
   827→      aiResponse = '❌ ব্যবসা নির্বাচন করুন। / Please select a business first.';
   828→    }
   829→
   830→    history.push({ role: 'assistant', content: aiResponse });
   831→    conversations.set(sessionId, history);
   832→
   833→    return NextResponse.json({ success: true, response: aiResponse, actionExecuted, actionData });
   834→  } catch (error) {
   835→    console.error('Chat API error:', error);
   836→    return NextResponse.json({ error: 'Failed to get response' }, { status: 500 });
   837→  }
   838→}
