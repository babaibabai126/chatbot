   226→   226→function extractClientName(message: string): string {
   227→   227→  const patterns = [
   228→   228→    /(?:for|client|customer|নাম|নামের)\s*:?\s*([A-Z][A-Za-z\s]{2,30}?)(?:\s*[,;.]|\s*(?:item|service|phone|email|with|amount|bill|create|add|₹)|$)/i,
   229→   229→    /(?:named?|name)\s*:?\s*([A-Z][A-Za-z\s]{2,30}?)(?:\s*[,;.]|\s*(?:phone|email|with|item|₹)|$)/i,
   230→   230→  ];
   231→   231→  for (const p of patterns) {
   232→   232→    const match = message.match(p);
   233→   233→    if (match) return match[1].trim();
   234→   234→  }
   235→   235→  return '';
   236→   236→}

   241→   241→  ];
   242→   242→  for (const p of patterns) {
   243→   243→    const match = message.match(p);
   244→   244→    if (match && match[1]) return match[1].trim();
   245→   245→  }
   246→   246→  const forMatch = message.match(/for\s+(.+?)\s+(?:₹|rs\.?|of)/i);
   247→   247→  if (forMatch) return forMatch[1].trim();

   251→   251→// ─── Process user's reply to a pending question ──
   252→   252→function processPendingReply(message: string, pending: PendingAction): DetectedIntent | null {
   253→   253→  const lower = message.toLowerCase().trim();
   254→   254→
   255→   255→  // If user is answering about bill type
   256→   256→  if (pending.missingFields.includes('billType')) {
   257→   257→    if (lower.includes('gst') && !lower.includes('non') && !lower.includes('নন')) {
   258→   258→      pending.args.billType = 'gst';
   259→   259→      pending.missingFields = pending.missingFields.filter(f => f !== 'billType');
   260→   260→    } else if (lower.includes('non') || lower.includes('নন') || lower === '1' || lower === 'non gst') {
   261→   261→      pending.args.billType = 'non_gst';
   262→   262→      pending.missingFields = pending.missingFields.filter(f => f !== 'billType');
   263→   263→    }
   264→   264→  }
   265→   265→
   266→   266→  // If user is answering about amount
   267→   267→  if (pending.missingFields.includes('amount')) {
   268→   268→    const amountMatch = lower.match(/[₹rs\s]*([\d,]+(?:\.\d+)?)/);
   269→   269→    if (amountMatch) {
   270→   270→      const amount = parseFloat(amountMatch[1].replace(/,/g, ''));
   271→   271→      if (amount > 0) {
   272→   272→        if (pending.action === 'create_bill') {
   273→   273→          pending.args.items = [{ itemName: 'Service', rate: amount, quantity: 1, taxMode: 'excl' }];
   274→   274→        } else if (pending.action === 'create_expense' || pending.action === 'record_payment') {
   275→   275→          pending.args.amount = amount;
   276→   276→        }
   277→   277→        pending.missingFields = pending.missingFields.filter(f => f !== 'amount');
   278→   278→      }
   279→   279→    }
   280→   280→  }
   281→   281→
   282→   282→  // If user is answering about client name
   283→   283→  if (pending.missingFields.includes('clientName')) {
   284→   284→    // Try to extract a name - must be actual name, not "GST bill" etc
   285→   285→    const nameMatch = message.match(/^([A-Z][A-Za-z\s]{1,30})$/);
   286→   286→    if (nameMatch && !nameMatch[1].toLowerCase().match(/^(gst|bill|non|service|expense)/)) {
   287→   287→      pending.args.clientName = nameMatch[1].trim();
   288→   288→      pending.missingFields = pending.missingFields.filter(f => f !== 'clientName');
   289→   289→    } else {
   290→   290→      // Try "for X" or "name X" patterns
   291→   291→      const forName = message.match(/(?:for|name[ds]?|নাম)\s*:?\s*([A-Z][A-Za-z\s]{1,30}?)(?:\s*[,;.]|$)/i);
   292→   292→      if (forName) {
   293→   293→        pending.args.clientName = forName[1].trim();
   294→   294→        pending.missingFields = pending.missingFields.filter(f => f !== 'clientName');
   295→   295→      }
   296→   296→    }
   297→   297→  }
   298→   298→
   299→   299→  // If user is answering about name (for client creation)
   300→   300→  if (pending.missingFields.includes('name')) {
   301→   301→    const nameMatch = message.match(/^([A-Z][A-Za-z\s]{1,30})$/);
   302→   302→    if (nameMatch) {
   303→   303→      pending.args.name = nameMatch[1].trim();
   304→   304→      pending.missingFields = pending.missingFields.filter(f => f !== 'name');
   305→   305→    } else {
   306→   306→      const forName = message.match(/(?:name[ds]?|নাম)\s*:?\s*([A-Z][A-Za-z\s]{1,30}?)(?:\s*[,;.]|$)/i);
   307→   307→      if (forName) {
   308→   308→        pending.args.name = forName[1].trim();
   309→   309→        pending.missingFields = pending.missingFields.filter(f => f !== 'name');
   310→   310→      }
   311→   311→    }
   312→   312→  }
   313→   313→
   314→   314→  // If user is answering about phone
   315→   315→  if (pending.missingFields.includes('phone')) {
   316→   316→    const phoneMatch = message.match(/([\d+\-\s]{8,15})/);
   317→   317→    if (phoneMatch) {
   318→   318→      pending.args.phone = phoneMatch[1].trim().replace(/\s/g, '');
   319→   319→      pending.missingFields = pending.missingFields.filter(f => f !== 'phone');
   320→   320→    }

   599→   599→    case 'record_payment': return executeRecordPayment(args, businessId);
   600→   600→    case 'get_dues': return executeGetDues(args, businessId);
   601→   601→    case 'get_dashboard': return executeGetDashboard(args, businessId);
   602→   602→    case 'list_clients': return executeListClients(args, businessId);
   603→   603→    case 'list_bills': return executeListBills(args, businessId);
   604→   604→    default: return { success: false, result: null, message: '❌ Unknown action' };
   605→   605→  }

   612→   612→export async function POST(request: NextRequest) {
   613→   613→  try {
   614→   614→    const { sessionId, message, context } = await request.json();
   615→   615→    if (!message || typeof message !== 'string') return NextResponse.json({ error: 'Message is required' }, { status: 400 });
   616→   616→
   617→   617→    const businessId = context?.businessId || '';
   618→   618→
   619→   619→    // Get or create conversation state
   620→   620→    let state = conversations.get(sessionId);
   621→   621→    if (!state) {
   622→   622→      state = { messages: [{ role: 'assistant', content: SYSTEM_PROMPT }], pendingAction: null };
   623→   623→    }
   624→   624→
   625→   625→    state.messages.push({ role: 'user', content: message });
   626→   626→    if (state.messages.length > 22) state.messages = [state.messages[0], ...state.messages.slice(-20)];
   627→   627→
   628→   628→    let aiResponse: string | null = null;
   629→   629→    let actionExecuted = false;
   630→   630→    let actionData = null;
   631→   631→
   632→   632→    // ── CHECK PENDING ACTION (multi-turn) ──
   633→   633→    if (state.pendingAction) {
   634→   634→      // Check if user is cancelling
   635→   635→      const lower = message.toLowerCase();
   636→   636→      if (lower.match(/cancel|বাতিল|না|no|stop|থাম/)) {
   637→   637→        state.pendingAction = null;
   638→   638→        aiResponse = '👌 বাতিল করা হয়েছে। অন্য কিছু বলুন! / Cancelled. What else can I help with?';
   639→   639→      } else {
   640→   640→        // Process the reply
   641→   641→        const updated = processPendingReply(message, state.pendingAction);
   642→   642→
   643→   643→        if (updated && updated.missingFields.length === 0) {
   644→   644→          // All info collected - execute!
   645→   645→          if (businessId) {
   646→   646→            const result = await executeFunction(updated.action, updated.args, businessId);
   647→   647→            actionExecuted = result.success;
   648→   648→            actionData = result.success ? result.result : null;
   649→   649→            aiResponse = result.message;
   650→   650→          } else {
   651→   651→            aiResponse = '❌ ব্যবসা নির্বাচন করুন।';
   652→   652→          }
   653→   653→          state.pendingAction = null;
   654→   654→        } else if (updated && updated.missingFields.length > 0) {
   655→   655→          // Still missing info - ask again
   656→   656→          state.pendingAction = { action: updated.action, args: updated.args, missingFields: updated.missingFields };
   657→   657→          aiResponse = askForMissing(state.pendingAction);
   658→   658→        } else {
   659→   659→          // Couldn't parse reply - ask again
   660→   660→          aiResponse = askForMissing(state.pendingAction);
   661→   661→        }
   662→   662→      }
   663→   663→    } else {
   664→   664→      // ── NEW INTENT DETECTION ──
   665→   665→      const intent = detectIntent(message);
   666→   666→
   667→   667→      if (intent) {
   668→   668→        if (intent.missingFields.length === 0) {
   669→   669→          // All info provided - execute immediately
   670→   670→          if (businessId) {
   671→   671→            const result = await executeFunction(intent.action, intent.args, businessId);
   672→   672→            actionExecuted = result.success;
   673→   673→            actionData = result.success ? result.result : null;
   674→   674→            aiResponse = result.message;
   675→   675→          } else {
   676→   676→            aiResponse = '❌ ব্যবসা নির্বাচন করুন।';
   677→   677→          }
   678→   678→        } else {
   679→   679→          // Missing info - ask user
   680→   680→          state.pendingAction = { action: intent.action, args: intent.args, missingFields: intent.missingFields };
   681→   681→          aiResponse = askForMissing(state.pendingAction);
   682→   682→        }
   683→   683→      } else {
   684→   684→        // No intent detected - use LLM for conversation
   685→   685→        const geminiText = await callGemini(state.messages);
   686→   686→        if (geminiText) {
   687→   687→          aiResponse = geminiText;
   688→   688→        } else {
   689→   689→          // Try ZAI
   690→   690→          try {
   691→   691→            const zai = await getZAI();
   692→   692→            if (zai) {
   693→   693→              const zaiMsgs = state.messages.filter(m => m.role === 'user' || (m.role === 'assistant' && m !== state!.messages[0])).slice(-8).map(m => ({ role: m.role === 'user' ? 'user' as const : 'assistant' as const, content: m.content }));
   694→   694→              zaiMsgs.unshift({ role: 'user', content: SYSTEM_PROMPT });
   695→   695→              const zaiResp = await zai.chat.completions.create({ messages: zaiMsgs, thinking: { type: 'disabled' } });
   696→   696→              aiResponse = zaiResp?.choices?.[0]?.message?.content || null;
   697→   697→            }
   698→   698→          } catch { /* ignore */ }
   699→   699→        }
   700→   700→
   701→   701→        // Final fallback
   702→   702→        if (!aiResponse) {
   703→   703→          const lower = message.toLowerCase();
   704→   704→          if (lower.match(/হ্যালো|হ্যাই|hello|hi|hey|নমস্কার|কেমন আছ/)) {
   705→   705→            aiResponse = `নমস্কার! / Hello! 👋\n\nআমি AAROHAN Business Hub এর AI সহকারী। আমি সরাসরি কাজ করতে পারি:\n\n• 📄 "বিল বানাও" / "Create a bill"\n• 👥 "ক্লায়েন্ট যোগ কর" / "Add client"\n• 💰 "খরচ লেখ" / "Record expense ₹500"\n• ⚠️ "বকেয়া দেখাও" / "Show dues"\n• 📊 "ড্যাশবোর্ড" / "Dashboard"\n\nকী করতে চান?`;
   706→   706→          } else {
   707→   707→            aiResponse = `🙏 আমি AAROHAN Business Hub এর AI সহকারী।\n\nআমি সরাসরি কাজ করতে পারি:\n• 📄 "বিল বানাও" বা "Create a GST bill for ₹10,000 for Rahul"\n• 👥 "ক্লায়েন্ট যোগ কর" বা "Add client named Priya, phone 9876543210"\n• 💰 "খরচ লেখ ₹500 internet" বা "Record expense ₹500 for internet"\n• ⚠️ "বকেয়া দেখাও" / "Show dues"\n• 📊 "ড্যাশবোর্ড"\n\nকী করতে চান?`;
   708→   708→          }
   709→   709→        }
   710→   710→      }
   711→   711→    }
   712→   712→
   713→   713→    state.messages.push({ role: 'assistant', content: aiResponse });
   714→   714→    conversations.set(sessionId, state);
   715→   715→
   716→   716→    return NextResponse.json({ success: true, response: aiResponse, actionExecuted, actionData });
   717→   717→  } catch (error) {
   718→   718→    console.error('Chat API error:', error);
   719→   719→    return NextResponse.json({ error: 'Failed to get response' }, { status: 500 });
   720→   720→  }
   721→   721→}
